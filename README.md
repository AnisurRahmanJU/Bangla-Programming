# বাংলা প্রোগ্রামিং

সিনট্যাক্স:
```py
ধরি নাম = "আনিছুর";
ধরি বয়স = 29;
লিখি(নাম + "এর বয়স " + বয়স);
```
কন্ডিশন: 

```py
ধরি নাম = "আকাশ"
যদি (নাম == "আনিছুর") {
লিখি("হ্যালো আনিছুর");
} নাহলে যদি (নাম == "লিখন") {
লিখি("হ্যালো লিখন");
} নাহলে {
লিখি("ঘোড়ার ডিম! তোমার নাম কি?");
}
```
while লুপ:
```py
ধরি ন = 1;
অব্দি(ন < 10) {
লিখি(ন);
ন = ন + 1;
}
```
